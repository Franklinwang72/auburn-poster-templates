% ---- the definition ----------------------------------------------------
\begin{frame}[plain]

  \substatement{A trivial canonical bundle forces $c_1(X) = 0$ — one whole number, free to compute.}
\end{frame}

\begin{frame}[plain]

% ---- the theorem -------------------------------------------------------
\begin{frame}[plain]
  \kicker{Yau, 1978 — the Calabi conjecture}
  \bigequation{\operatorname{Ric}(\tilde\omega) \;=\; 0}
  \substatement{A unique Ricci-flat metric — one in every Kähler class.}
\end{frame}

\begin{frame}[plain]
  \kicker{How?}

   $\bigl(\omega + i\,\partial\bar\partial\varphi\bigr)^{n} = e^{F}\omega^{n}$\par}
  \substatement{The complex Monge–Ampère equation, solved by a priori estimates.}
\end{frame}
